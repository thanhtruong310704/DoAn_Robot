#include "debug.h"

ControlRobot ::ControlRobot(MotorStep& m1, MotorStep& m2, MotorStep& m3)
    : step1(m1),
      step2(m2),
      step3(m3),
      state(WAIT),
      running_traj(false),
      t_traj(0),
      test(false) {}
void ControlRobot :: init() {
  step1.as5600begin();
  step1.begin();
  step2.as5600begin();
  step2.begin();
  step3.as5600begin();
  step3.begin();
  setupSetPoint();
}


void ControlRobot :: update(float denta_t) {
  step1.goc = step1.readAngle();
  step2.goc = step2.readAngle() -90 ;
  step3.goc = -step3.readAngle() - step2.goc;
  step1.output = step1.pid.outputPID(denta_t, step1.setpoint, step1.goc, step1.Kp, step1.Ki, step1.Kd);  //4.7, 0.001, 0.7
  step2.output = step2.pid.outputPID(denta_t, step2.setpoint, step2.goc, step2.Kp, step2.Ki, step2.Kd);
  step3.output = step3.pid.outputPID(denta_t, step3.setpoint, step3.goc, step3.Kp, step3.Ki, step3.Kd);
}

void ControlRobot :: getDataCV2() {
  checkSerial();
  planTrajectory();
}


void ControlRobot :: Run() {

  step1.output = constrain(step1.output, -step1.step.maxSpeed(), step1.step.maxSpeed());
  step1.step.setSpeed(-step1.output *5);
  step2.output = constrain(step2.output, -step2.step.maxSpeed(), step2.step.maxSpeed());
  step2.step.setSpeed(step2.output*5);
  step3.output = constrain(step3.output, -step3.step.maxSpeed(), step3.step.maxSpeed());
  step3.step.setSpeed(-step3.output*5);

  if ((fabs(step1.output) > 1) and (fabs(step2.output) > 1) and (fabs(step2.output) > 1)){
    step1.step.runSpeed();
  }
  else if ((fabs(step2.output) > 1) and (fabs(step2.output) > 1)) {
    step2.step.runSpeed();
    step3.step.runSpeed();
  }

  #ifdef DEBUG_OUT
    Serial.print("q1: "); Serial.println(step1.pid.out);
    Serial.print("q2: "); Serial.println(step2.pid.out);
    Serial.print("q3: "); Serial.println(step3.pid.out);
  #endif
}

void  ControlRobot :: setupSetPoint() {

  step1.setpoint = step1.readAngle();
  step2.setpoint = step2.readAngle() -90;
  step3.setpoint = -step3.readAngle() - step2.setpoint ;

}

void ControlRobot :: checkSerial() {
  if (Serial.available() > 0) {
    String data = Serial.readStringUntil('\n');
    data.trim();
    int firstComma = data.indexOf(',');
    int secondComma = data.indexOf(',', firstComma + 1);
    if (firstComma > 0 && secondComma > 0) {
      finish.x = data.substring(0, firstComma).toFloat();
      finish.y = data.substring(firstComma + 1, secondComma).toFloat();
      finish.z = data.substring(secondComma + 1).toFloat();
    }
    test=true;
    #ifdef  DEBUG_CV2
      Serial.print("x= ");
      Serial.print(finish.x);
      Serial.print(" y= ");
      Serial.print(finish.y);
      Serial.print(" z=");
      Serial.println(finish.z);
    #endif
  }
}


void ControlRobot ::  planTrajectory(){
  if (test) {
    kinematic.IK(finish.x, finish.y, finish.z, step1.setpoint, step2.setpoint, step3.setpoint);
    test = false;
  }
}
